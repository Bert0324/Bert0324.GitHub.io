var reverse = function(x) {
    const s = Array.from(`${x}`);
    let minus = false;
    if (s[0] === '-') {
        s.shift();
        minus = true;
    }
    s.reverse();
    const v = Number(s.join(''));
    if (v > 2 ** 31 - 1) return 0;
    return minus ? -v : v;
};
